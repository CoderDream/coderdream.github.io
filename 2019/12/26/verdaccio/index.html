查询端口是否开放

firewall-cmd –query-port=4873/tcp

开放4873端口

firewall-cmd –permanent –add-port=4873/tcp

移除端口

firewall-cmd –permanent –remove-port=4873/tcp

#重启防火墙(修改配置后要重启防火墙)
firewall-cmd –reload

nrm use S229
